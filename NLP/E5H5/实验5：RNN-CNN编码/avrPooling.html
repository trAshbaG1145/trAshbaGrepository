import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import fetch_20newsgroups
from sklearn.model_selection import train_test_split
from gensim.models import Word2Vec
from sklearn.decomposition import PCA
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import classification_report, accuracy_score

# 20 Newsgroups
newsgroups = fetch_20newsgroups(subset='all', categories=['alt.atheism', 'soc.religion.christian'], remove=('headers', 'footers', 'quotes'))

# DataFrame
data = pd.DataFrame({'text': newsgroups.data, 'label': newsgroups.target})

# 
train_data, test_data = train_test_split(data, test_size=0.2, random_state=42)

# 
train_sentences = [text.split() for text in train_data['text']]
test_sentences = [text.split() for text in test_data['text']]

# Word2Vec
word2vec_model = Word2Vec(sentences=train_sentences, vector_size=100, window=5, min_count=1, workers=4)

# 
def get_text_vector(text, model):
    words = text.split()
    word_vectors = [model.wv[word] for word in words if word in model.wv]
    if not word_vectors:
        return np.zeros(model.vector_size)  # 
    return np.mean(word_vectors, axis=0)

# 
train_data['vector'] = train_data['text'].apply(lambda x: get_text_vector(x, word2vec_model))
test_data['vector'] = test_data['text'].apply(lambda x: get_text_vector(x, word2vec_model))

# 
X_train = np.array(train_data['vector'].tolist())
y_train = train_data['label'].values
X_test = np.array(test_data['vector'].tolist())
y_test = test_data['label'].values

# 
logistic_model = LogisticRegression(max_iter=1000)
logistic_model.fit(X_train, y_train)

# 
y_pred = logistic_model.predict(X_test)

# 
accuracy = accuracy_score(y_test, y_pred)
report = classification_report(y_test, y_pred)

print(f'Accuracy: {accuracy:.2f}')
print('Classification Report:')
print(report)

# PCA2D
pca = PCA(n_components=2)
X_train_2d = pca.fit_transform(X_train)

# 
plt.figure(figsize=(10, 6))
scatter = plt.scatter(X_train_2d[:, 0], X_train_2d[:, 1], c=y_train, cmap='coolwarm', alpha=0.7)
plt.title('Word2Vec Text Representation with Logistic Regression')
plt.xlabel('PCA Component 1')
plt.ylabel('PCA Component 2')
plt.colorbar(scatter, ticks=[0, 1], label='Class')
plt.grid()
plt.show()
